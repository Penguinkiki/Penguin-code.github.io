F2 六個數字的排序

  1. 產生一組抽獎號
  2. 將取亂數的程 (1-49) ,加上排序功能 (由小到大)
  3. 將排序前後的陣列(result) 結果印出來(要用陣列/串列儲存)
def bubbleSort(arr): #宣告氣泡排序法
    for i in range(1, len(arr)): #執行次數
        for j in range(0, len(arr)-i): #list的編號
            if arr[j] > arr[j+1]: #如果後數字比前數字大，就交換
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
    return arr
from random import * #import random函式
a = [] #宣告list
for i in range(6):
    a.append(randint(1,49)) #插入六次隨機1-49整數
print(a) #輸出排序前的list
print(bubbleSort(a))

a = []

print(a)
